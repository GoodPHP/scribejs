Quick Start

One-line initialization and a simple React wrapper.

import { Scribe } from './scribe.js';

// Initialize with one line
const editor = Scribe.init('#editor');

// Direct methods — no exec() needed!
editor.bold();
editor.italic();
editor.link('https://example.com');
editor.heading(2);

// Get content
const html = editor.getHTML();
import { ScribeEditor } from '@scribejs/react';

function App() {
  const editorRef = useRef(null);
  
  return (
    <ScribeEditor
      ref={editorRef}
      defaultValue="<p>Start writing...</p>"
      toolbar="floating"
      onChange={(html) => console.log(html)}
    />
  );
}

Simple, Intuitive API

editor.bold()

Direct methods for formatting — no command strings to memorize.

editor.italic() | editor.underline()

editor.heading(2)

Block formatting with simple, typed parameters.

editor.blockquote() | editor.orderedList()

editor.link(url)

Insert and manage links with ease.

editor.link('https://...') | editor.unlink()

editor.on('change', fn)

Subscribe to events for real-time updates.

'change' | 'focus' | 'blur' | 'selectionChange'
